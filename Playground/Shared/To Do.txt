Better Error Return Macro
	MACRO(expression) => can log the expression with L##expression

	#define IGNORE
	#define FALSE !

	#define IF(expression, check, action) \
	do {                                  \
		if ( check(expression) )          \
		{                                 \
			action;                       \
		}                                 \
	} while ( 0 )
